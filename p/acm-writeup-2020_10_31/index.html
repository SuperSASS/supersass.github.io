struct Data
{
	int h, v;
	bool operator<(const Data &t) { return v - h > t.v - t.h; }
};

int n;
Data d[N];

int main()
{
	scanf("%d", &n);
	for (int i = 1; i <= n; ++i)
		scanf("%d%d", &d[i].h, &d[i].v);
	sort(d + 1, d + n + 1);
	int tmp = 0, res = 1e9;
	for (int i = 1; i <= n; ++i)
	{
		tmp += d[i].v;
		res = min(res, tmp - d[i].h);
	}
	printf("%d\n", res);
	return 0;
}
